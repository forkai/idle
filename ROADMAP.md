## 🤖 AI队友/酒馆系统（核心创新）

### AI队友系统设计

#### 角色设定
```typescript
interface AICompanion {
  id: string
  name: string                    // 角色名称
  avatar: string                  // 头像emoji

  // 性格系统
  personality: {
    trait: PersonalityTrait        // 性格类型
    mood: number                  // 当前心情 (0-100)
    attitude: Attitude             // 对玩家态度
    speakingStyle: SpeakingStyle  // 说话风格
  }

  // 记忆系统
  memory: {
    episodic: Episode[]            // 情景记忆（战斗、事件）
    semantic: Knowledge[]         // 知识记忆（游戏知识）
    emotional: EmotionalEvent[]   // 情感记忆
  }

  // 能力系统
  abilities: {
    combatPower: number           // 战斗能力
    supportBonus: number          // 支援加成
    unlockCondition: string        // 解锁条件
  }

  // 好感度系统
  relationship: {
    affinity: number              // 好感度
    trust: number                 // 信任度
    storyProgress: number         // 剧情进度
  }
}

// 性格类型
type PersonalityTrait =
  | '热血'    // 战斗狂，喜欢冲在前线
  | '冷静'    // 分析型，战斗策略型
  | '乐观'    // 开朗，话多，正能量
  | '腹黑'    // 毒舌，但内心善良
  | '高冷'    // 话少，但关键时靠谱
  | '话痨'    // 什么都喜欢说
```

#### 记忆系统
```typescript
interface CompanionMemory {
  // 情景记忆 - 具体的经历
  episodes: {
    id: string
    type: 'combat' | 'event' | 'conversation' | 'milestone'
    content: string
    timestamp: number
    emotional: 'positive' | 'negative' | 'neutral'
    relatedEntities: string[]   // 相关实体
  }[]

  // 知识记忆 - 学到的游戏知识
  knowledge: {
    topic: string                // 话题
    facts: string[]              // 已知事实
    lastUpdated: number
  }[]

  // 偏好记忆 - 玩家偏好
  preferences: {
    playStyle: 'aggressive' | 'defensive' | 'balanced'
    favoriteZone: string
    favoriteEquipment: string[]
    petPeeves: string[]          // 雷点
  }
}

// 记忆处理
interface MemoryConfig {
  maxEpisodic: 100                // 最多100条情景记忆
  maxSemantic: 50                 // 最多50条知识
  consolidationThreshold: 10      // 记忆整合阈值
  decayRate: 0.01                // 记忆衰减率
}
```

#### 对话系统
```typescript
// AI酒馆对话
interface DialogueSystem {
  // 对话上下文
  context: {
    currentLocation: string       // 当前位置
    recentEvents: string[]        // 最近事件
    playerState: PlayerSummary     // 玩家状态摘要
  }

  // 可用回复选项
  responseOptions: {
    text: string                  // 回复文本
    sentiment: number             // 情感值
    triggersMemory: string[]       // 触发的记忆
    affinityChange: number         // 好感度变化
  }[]

  // LangChain集成点
  chain: {
    prompt: PromptTemplate
    memory: VectorStoreRetriever
    outputParser: DialogueParser
  }
}
```

## 🙏 Sprint 23-25: 神主系统（20-25小时）

> 玩家角色达到最高等级（Lv.99）后可在魔王和神主之间二选一转职。
> 神主是魔王的镜像玩法：从攻击者变为祝福者，通过恩赐系统帮助勇者变强。

### 神主 vs 魔王 对比
| 维度 | 魔王 | 神主 |
|------|------|------|
| 定位 | 地下城设计者，攻击方 | 信仰授予者，守护方 |
| 核心行为 | 部署怪物挑战勇者 | 颁布恩赐任务奖励勇者 |
| 目标 | 击杀更多勇者获取经验 | 吸引信徒完成任务获取经验 |
| 升级方式 | 杀死勇者越多越强 | 信徒越多、任务越完成越强 |

```typescript
interface DivineBlessing {
  id: string
  godId: string                  // 所属神主
  name: string                  // 恩赐名称（如"圣光庇护"）
  description: string           // 描述
  taskRequirements: TaskGoal[]   // 任务目标
  rewards: RewardPool           // 奖励池
  cooldown: number               // 冷却时间（ms）
  maxCompletions: number        // 每日最大完成次数
  faithRequired: number         // 需要虔诚值门槛
}

interface TaskGoal {
  type: 'kill_monster' | 'collect_item' | 'clear_zone' | 'defeat_dungeon' | 'gather_faith'
  targetId: string
  targetCount: number
  progress: number               // 当前进度
  completed: boolean
}

interface GodMaster {
  playerId: string
  godLevel: number              // 神主等级（独立）
  title: string                 // 神主称号
  totalBelievers: number        // 累计信徒数
  activeBelievers: number       // 当前活跃信徒数
  divinePower: number            // 神力值（影响任务奖励幅度）
  blessings: DivineBlessing[]   // 已发布的恩赐
  tier: number                  // 神主层级（1-5，越高能力越强）
}
```
